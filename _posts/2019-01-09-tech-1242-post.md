    <?php
    
    <br></br>
    
    use Illuminate\Database\Seeder;
    
    <br></br>
    
    class SimpleCommentSeeder extends Seeder
    
    {
    
        /**
    
         * Run the database seeds.
    
         *
    
         * @return void
    
         */
    
        public function run()
    
        {
    
          $sc = new App\SimpleComment();
    
          for($i=0,$m=20;$i<$m;$i++){
    
            $t = str_repeat(rand(0,9), 4);
    
            $sc->create([
    
              // 'sc_id'=>null,
    
              'sc_name'=>$t,
    
              'sc_pass'=>$t,
    
              'sc_comment'=>md5(rand(1,9))." ".md5(rand(1,9)),
    
            ]);
    
    
    
          }
    
        }
    
    
    
    }

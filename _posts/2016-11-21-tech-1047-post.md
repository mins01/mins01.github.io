[레퍼런스링크](http://api.jquery.com/jquery.ajax/ "레퍼런스링크")  


$.ajax({

	url: url,

	type: 'POST', //GET

	dataType: 'json', //xml, json, script, jsonp, or html

	data: post_data,

})

.done(function(rData) { //통신 성공 시 호출

	console.log("success");

	

})

.fail(function() { //통신 실패 시 호출

	console.log("error");

})

.always(function() { //성공/실패 후 호출.

	console.log("complete");

});

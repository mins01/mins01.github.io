#!/bin/bash

  


YMD=$(date +%Y%m%d%H%M%S)

  


array=($(sh ../count_listener.sh|awk '{print $2}'))  #명령을 배열로 만듬

  


cnt=${#array[@]} # 배열 갯수

  


for str in "${array[@]}"  #배열 루프

do

        echo "${str}" &gt;&gt; status.${YMD}.log

        ../../bin/sqlr-status -id ${str} | grep Total &gt;&gt; status.${YMD}.log

done
